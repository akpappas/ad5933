877 //Execute frequency sweep.
878 // lower: Starting frequency of the sweep.
879 // number_of_samples: Frequencies measured in the sweep
880 // step: Distance between frequencies meausured in the sweep.
881 // h: Handle to the device object.
882 //The function implements the flowchart on page 20 of the data sheet.
883 std::vector< std::pair<long double, complex_t > > sweep_frequency ( uint32_t lower,uint32_t number_of_samples,long double step, AD5933* h )
884 {
885  long double clk = h->clk;
886  vector< pair<long double,complex_t>> measurements;
887  long double lowerd= lower;
888  auto startd = (lowerd / (clk/4))* (1<<27);
889  auto incd = (step / (clk/4))*(1<<27);
890 
891  uint32_t start = startd;
892  uint32_t inc = incd;
893  h->set_starting_frequency ( start );
894  h->set_frequency_step ( inc );
895  h->set_step_number ( number_of_samples );
896 
897  h->set_standby();
898  usleep ( 5e5 );
899 
900  h->initilize_frequency();
901  usleep ( 5e5 );
902 
903  h->start_sweep();
904  usleep( 5e5);
905 
906 #ifdef DEBUG
907  std::cout<<"clock "<<clk<<"\n";
908  std::cout<<"lowerd:"<<lowerd<<" "<<lower<<std::endl;
909  printf("lower %d start = 0x%X,inc = 0x%X\n",lower,start,inc);
910  printf("Set frequency: 0x%X\n",h->get_frequency());
911 #endif
912 
913  /*Sweep loop*/
914  long double true_freq;
915  auto cur_freq = start;
916  int i=0;
917  uint8_t sreg;
918  for ( ;; )
919  {
920  long double ar = ( cur_freq );
921  true_freq = ar/ ( (1<<27)/(clk/4)) ;
922  /*Read SREG for valid impedance meausurement*/
923  uint8_t impedance_valid=0;
924  std::bitset<8> sregB(h->get_status());
925  // std::cout<<sregB<< "sreg before\n";
926  do
927  {
928  sreg = h->get_status();
929  impedance_valid = sreg & SREG_IMPED_VALID;
930  }
931  while ( !impedance_valid );
932  sregB = h->get_status();
933  // std::cout<<sregB<<"sreg after"<<std::endl;
934  auto z = h->read_measurement();
935  measurements.push_back ( make_pair ( true_freq,z ) );
936  cur_freq+=inc;
937  i++;
938  sreg = h->get_status();
939  if ( sreg & SREG_SWEEP_VALID ) break;

941  }

1027 //Simple interpolation function.
1028 // f: the intermediate point where the interpolated value lies
1029 // g0: the first existing data point
1030 // g1: the second existing data point
1031 //Returns the interpolated value.
1032 long double interpolate(long double f,
1033  const std::pair<long double, long double> &g0,
1034  const std::pair<long double, long double> &g1)
1035 {
1036  long double new_gain = 0;
1037  if (std::abs(g0.first-f)<0.1)
1038  {
1039  new_gain = g0.second;

1042  {
1043  new_gain = g1.second;
1044  }
1045  else
1046  {
1047  auto frac = (f-g0.first)/(g1.first-g0.first);
1048  new_gain = (1-frac)*g0.second + frac*g1.second;

1053 //Interpolate the gain factors based on the calibration data.
1054 // adm: reference to the vector containing the frequency, gain pairs to be calcualted
1055 // cal: reference to the vector containing to the known frequency, gain pairs.
1056 template <typename T>
1057 std::vector<std::pair<long double, long double>>
1058 calc_multigains(const std::vector<std::pair<long double, T>> &adm,
1059  const std::vector<std::pair<long double, long double>> &cal)
1060 {
1061  std::vector< std::pair<long double, long double>> new_g;
1062  // Gains were calibrated at higher frequencies, use the gain calibrated at the lowest frequency
1063  size_t k=0;
1064  while (k<adm.size() && adm[k].first<cal[0].first)
1065  {
1066  new_g.push_back(std::make_pair(adm[k].first,cal[0].second));
1067  k++;
1068  }
1069  // Gains were calibrated at frequencies both higher and lower frequencies. Interpolate between them.
1070  for (size_t i=0;i<cal.size()-1;i++)
1071  {
1072  while (k<adm.size() && adm[k].first<cal[i+1].first)
1073  {
1074  auto temp = interpolate(adm[k].first, cal[i],cal[i+1]);
1075  new_g.push_back(std::make_pair(adm[k].first,temp));
1076  k++;
1077  }
1078  }
1079  //Gains were calibrated at lower frequencies, use the gain calibrated at the highest frequency.
1080  while (k<adm.size())
1081  {
1082  new_g.push_back(std::make_pair(adm[k].first,cal.back().second));
1083  k++;
1084  }
1085  return new_g;
